import cv2
import numpy as np
import matplotlib.pyplot as plt

img_path = 'rose.png'  
img_bgr = cv2.imread(img_path)

if img_bgr is None:
    print(f":  {img_path}  !  .")
else:
    img_gray = cv2.cvtColor(img_bgr, cv2.COLOR_BGR2GRAY)

    def plot_img_and_hist(image, title, is_gray=True):
        fig, axes = plt.subplots(1, 2, figsize=(12, 4))
        
        if is_gray:
            axes[0].imshow(image, cmap='gray', vmin=0, vmax=255)
            axes[1].hist(image.ravel(), bins=256, range=[0, 256], color='black')
        else:
            axes[0].imshow(image)
            colors = ('r', 'g', 'b')
            for i, color in enumerate(colors):
                hist = cv2.calcHist([image], [i], None, [256], [0, 256])
                axes[1].plot(hist, color=color)
                
        axes[0].set_title(f': {title}')
        axes[0].axis('off')
        axes[1].set_title(f': {title}')
        axes[1].set_xlim([0, 256])
        plt.show()

    plot_img_and_hist(img_gray, " ( )")
